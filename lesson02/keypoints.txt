-----   commands    -----

py --version                                    // check version of python
py -m venv "<name-of-venv>"                     // create virtual environment
venv\Scripts\activate                           // activate venv in PS
py -m pip install django                        // install django in venv
py -m pip install -U pip                        // use "-U" for "--upgrade"
django-admin startproject <project-name>        // create django project
py manage.py runserver                          // run server
