py manage.py createsuperuser                    // create a super user

-----   Templating Engine   -----

{% load static %}                               // import static files
href="{% static 'css/style.css' %}"             // apply css to html
src="{% static 'js/main.js' %}" defer           // apply js to html
{% block <bolck-name:title> %}                  // start a block
{% endblock <bolck-name:title> %}               // end the block
{% extends <html-page:layout.html> %}           // enables code reusability
{% for <item:post> in <items:posts> %}          // start for loop in html using python
{% endfor %}                                    // end for loop
{{ post.title }}                                // what item
{% url "<url-name:list>" %}                     // dynamically render urls
src="{{ post.banner.url }}"                     // render image
